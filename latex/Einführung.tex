    Eine komplexe Zahl $\mathbf{z \in  \mathbb{C}}$ ist \textbf{ein Wertepaar} $\mathbf{(a; b)}$, wobei $\mathbf{a}$ den \textbf{Realenteil} und $\mathbf{b}$ den \textbf{Imagin√§renteil} darstellt.

    \begin{tabular}{c|c}
        \parbox{0.45\textwidth}
        {
            \begin{alignat*}{3}
                &z &&= \left(a; b\right) \qquad z \in \mathbb{C} \\ 
                &a &&= \mbox{Re}(z) \\ 
                &b &&= \mbox{Im}(z) \\
            \end{alignat*}
        }
        &
        \parbox{0.45\textwidth}
        {
            \begin{alignat*}{3}
                &\left(a; b\right) + \left(c; d\right) &&= \left(a + c; b + d\right) \\
                &\left(a; b\right) - \left(c; d\right) &&= \left(a - c; b - d\right) \\
                &\left(a; b\right) \cdot \left(c; d\right) &&= \left(ac - bd; ad + bc\right) \\
                &\,\,r \cdot \left(a; b\right) &&= \left(ra; rb\right) \qquad r \in \mathbb{R} \\
            \end{alignat*}
        }
    \end{tabular}
}
